三个水杯倒水问题——JAVA实现(001)

Oct 12, 2017 2 min read

题目：给定一个装满水的8升杯子以及两个容量分别为5升和3升的空杯子。如何通过互相倒空或装满这些水杯，经过最少的操作得到4升的水。

首先，这道题要求通过最少的步骤获得解，因此可以通过广度优先搜索的方式进行搜索，每次的搜索都会有6种不同的倒水方式，在搜索的过程中通过设置一些条件(如需要倒水的杯子为空，目标的杯子已经满了等)判断是否可以完成倒水，可以裁剪掉大量不符合实际的倒水方式。同时维护一个hashmap，将每种新出现的倒水方案加入其中，这样可以避免很多重复的情形出现。

图1 倒水过程状态图

完整的代码如下，关键部分已加了注释

import java.util.HashMap;
import java.util.LinkedList;

public class PourWater {
    public static void main(String[] args) {
        LinkedList<State> states = new LinkedList<State>();     //广度优先搜索过程保存待访问节点的列表(FIFO)
        HashMap<Integer, State> history = new HashMap<Integer, State>();    //history中保存每种新出现的状态
        State state = new State(8, 0, 0);       //初始化状态为8,0,0
        states.addLast(state);
        history.put(state.getNum(), state);
        boolean hasFind = false;
        while (states.size() > 0||!hasFind) {
            State tmp = states.removeFirst();   //FIFO
            for (int i = 0; i < 3; i++)   //pour water from i to j
            {
                for (int j = 0; j < 3; j++) {       //每个节点都有6种状态(包括不可行的状态)
                    if (i != j) {
                        if (tmp.canPour(i, j)) {
                            State newState = new State(tmp.num);
                            newState.pour(i, j);
                            newState.preState = tmp;
                            if (!history.containsKey(newState.getNum())) {      //如果这种状态没有出现过，则将这种状态加入history中
                                history.put(newState.getNum(), newState);
                                states.addLast(newState);
                            }
                            if (newState.num[0] == 4 && newState.num[1] == 4) {     //如果达成条件则打印整个倒水过程
                                while (newState.preState!=null)
                                {
                                    System.out.println(newState.toString());
                                    newState = newState.preState;
                                }
                                System.out.println(newState.toString());
                                hasFind = true;
                                break;
                            }
                        }
                    }
                }
                if (hasFind)
                    break;
            }
        }
        if (!hasFind)
            System.out.print("No such solution!");
    }

}

class State {
    int[] num = new int[3];        //每个杯子中水的数量   
    int[] capicity = {8, 5, 3};   //每个杯子的最大容量
    State preState;     //  preState中保存着当前状态的前一个状态，这样当找到最终倒水方案时可以通过不断访问preState获得完整的倒水过程

    public State(int x1, int x2, int x3) {
        this.num[0] = x1;
        this.num[1] = x2;
        this.num[2] = x3;
    }

    public State(int[] x) {
        this.num[0] = x[0];
        this.num[1] = x[1];
        this.num[2] = x[2];
    }

    // 判断从from到to的倒水是否可行
    public boolean canPour(int from, int to) {
        if (num[from] == 0)
            return false;
        if (num[to] == capicity[to])
            return false;
        else
            return true;
    }

    //将水从from倒入to中
    public void pour(int from, int to) {
        if (num[from] + num[to] > capicity[to]) {
            num[from] -= (capicity[to] - num[to]);
            num[to] = capicity[to];
        } else {
            num[to] += num[from];
            num[from] = 0;
        }
    }

    public int getNum() {
        return 100 * num[0] + 10 * num[1] + num[2];
    }
    public String toString()
    {
        return "cup1: "+num[0]+" cup2: "+num[1]+" cup3: "+num[2];
    }
}

最终，我们的输出如下： cup1: 4 cup2: 4 cup3: 0 cup1: 1 cup2: 4 cup3: 3 cup1: 1 cup2: 5 cup3: 2 cup1: 6 cup2: 0 cup3: 2 cup1: 6 cup2: 2 cup3: 0 cup1: 3 cup2: 2 cup3: 3 cup1: 3 cup2: 5 cup3: 0 cup1: 8 cup2: 0 cup3: 0
